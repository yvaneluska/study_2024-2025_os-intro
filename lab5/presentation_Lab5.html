# Менеджер паролей pass

- стандартный менеджер паролей Unix
- хранение паролей в файловой системе
- шифрование с помощью GPG

# Основные команды

pass insert site/user

pass site/user

pass generate site/user

# Синхронизация

pass git init  
pass git push  
pass git pull
